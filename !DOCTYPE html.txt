        // Calculate stock status
        function getStockStatus(total) {
            if (total === 0) return 'CRITICAL';
            if (total <= 5) return 'LOW';
            return 'GOOD';
        }

        // Get status class
        function getStatusClass(status) {
            switch(status) {
                case 'CRITICAL': return 'status-critical';
                case 'LOW': return 'status-low';
                case 'GOOD': return 'status-good';
                default: return '';
            }
        }

        // Calculate product total
        function calculateProductTotal(product) {
            let total = 0;
            Object.keys(stockData[product]).forEach(size => {
                const item = stockData[product][size];
                total += item.godown1 + item.godown2 + item.godown3;
            });
            return total;
        }

        // Adjust stock
        function adjustStock(product, size, godown, change) {
            console.log(`Adjusting stock: ${product}, ${size}, ${godown}, ${change}`);
            const currentValue = stockData[product][size][godown];
            const newValue = Math.max(0, currentValue + change);
            stockData[product][size][godown] = newValue;
            
            renderStockTables();
            updateSummaryCards();
            updateCharts();
            updateLastUpdated();
        }

        // Quick stock actions
        function quickAddStock() {
            console.log('Adding stock...');
            const product = document.getElementById('quickProduct').value;
            const size = document.getElementById('quickSize').value;
            const godown = document.getElementById('quickGodown').value;
            const quantity = parseInt(document.getElementById('quickQuantity').value) || 0;

            if (product && size && godown && quantity > 0) {
                const godownKey = `godown${godown}`;
                stockData[product][size][godownKey] += quantity;
                
                renderStockTables();
                updateSummaryCards();
                updateCharts();
                updateLastUpdated();
                
                document.getElementById('quickQuantity').value = '';
            } else {
                alert('कृपया सभी fields भरें और valid quantity डालें');
            }
        }

        function quickSubtractStock() {
            console.log('Subtracting stock...');
            const product = document.getElementById('quickProduct').value;
            const size = document.getElementById('quickSize').value;
            const godown = document.getElementById('quickGodown').value;
            const quantity = parseInt(document.getElementById('quickQuantity').value) || 0;

            if (product && size && godown && quantity > 0) {
                const godownKey = `godown${godown}`;
                stockData[product][size][godownKey] = Math.max(0, stockData[product][size][godownKey] - quantity);
                
                renderStockTables();
                updateSummaryCards();
                updateCharts();
                updateLastUpdated();
                
                document.getElementById('quickQuantity').value = '';
            } else {
                alert('कृपया सभी fields भरें और valid quantity डालें');
            }
        }

        // Update summary cards
        function updateSummaryCards() {
            console.log('Updating summary cards...');
            let totalStock = 0;
            let lowStockItems = 0;
            let criticalStockItems = 0;

            Object.keys(stockData).forEach(product => {
                Object.keys(stockData[product]).forEach(size => {
                    const item = stockData[product][size];
                    const total = item.godown1 + item.godown2 + item.godown3;
                    totalStock += total;
                    if (total === 0) criticalStockItems++;
                    else if (total <= 5) lowStockItems++;
                });
            });

            document.getElementById('totalStock').textContent = totalStock;
            document.getElementById('lowStockItems').textContent = lowStockItems;
            document.getElementById('criticalStock').textContent = criticalStockItems;
        }

        // Create charts
        let productChartInstance = null;
        let godownChartInstance = null;

        function createCharts() {
            console.log('Creating charts...');
            createProductChart();
            createGodownChart();
        }

        function createProductChart() {
            console.log('Creating product chart...');
            if (productChartInstance) productChartInstance.destroy();

            const productTotals = Object.keys(stockData).map(product => ({
                product,
                total: calculateProductTotal(product)
            }));

            productChartInstance = new Chart(document.getElementById('productChart'), {
                type: 'bar',
                data: {
                    labels: productTotals.map(item => item.product),
                    datasets: [{
                        label: 'Total Stock',
                        data: productTotals.map(item => item.total),
                        backgroundColor: '#3B82F6',
                        borderColor: '#2563EB',
                        borderWidth: 1
                    }]
                },
                options: {
                    maintainAspectRatio: false,
                    scales: {
                        y: { beginAtZero: true }
                    }
                }
            });
        }

        function createGodownChart() {
            console.log('Creating godown chart...');
            if (godownChartInstance) godownChartInstance.destroy();

            const godownTotals = {
                godown1: 0,
                godown2: 0,
                godown3: 0
            };

            Object.keys(stockData).forEach(product => {
                Object.keys(stockData[product]).forEach(size => {
                    const item = stockData[product][size];
                    godownTotals.godown1 += item.godown1;
                    godownTotals.godown2 += item.godown2;
                    godownTotals.godown3 += item.godown3;
                });
            });

            godownChartInstance = new Chart(document.getElementById('godownChart'), {
                type: 'pie',
                data: {
                    labels: ['Godown 1', 'Godown 2', 'Godown 3'],
                    datasets: [{
                        data: [godownTotals.godown1, godownTotals.godown2, godownTotals.godown3],
                        backgroundColor: ['#3B82F6', '#10B981', '#F59E0B'],
                        borderColor: '#FFFFFF',
                        borderWidth: 2
                    }]
                },
                options: {
                    maintainAspectRatio: false
                }
            });
        }

        function updateCharts() {
            console.log('Updating charts...');
            createProductChart();
            createGodownChart();
        }

        // Export to Excel
        function exportToExcel() {
            console.log('Exporting to Excel...');
            const wb = XLSX.utils.book_new();
            const ws_data = [['Product', 'Size', 'Godown 1', 'Godown 2', 'Godown 3', 'Total', 'Status']];

            Object.keys(stockData).forEach(product => {
                Object.keys(stockData[product]).forEach(size => {
                    const item = stockData[product][size];
                    const total = item.godown1 + item.godown2 + item.godown3;
                    const status = getStockStatus(total);
                    ws_data.push([product, size, item.godown1, item.godown2, item.godown3, total, status]);
                });
            });

            const ws = XLSX.utils.aoa_to_sheet(ws_data);
            XLSX.utils.book_append_sheet(wb, ws, 'Stock Report');
            XLSX.writeFile(wb, 'Stock_Report.xlsx');
        }
